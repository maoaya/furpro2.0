# Comandos para configurar Git y subir tu proyecto

# 1. Configurar Git (reemplaza con tu info)
git config --global user.name "Tu Nombre"
git config --global user.email "tu.email@ejemplo.com"

# 2. Inicializar repositorio en tu proyecto
cd "c:\Users\lenovo\Desktop\futpro2.0"
git init

# 3. Crear .gitignore
echo "node_modules/
.env
.env.local
.env.production
dist/
.vscode/
.DS_Store
*.log
coverage/" > .gitignore

# 4. Agregar archivos
git add .

# 5. Hacer primer commit
git commit -m "Initial commit: FutPro 2.0 with futpro.vip domain"

# 6. Conectar con GitHub (reemplaza USERNAME con tu usuario de GitHub)
git remote add origin https://github.com/USERNAME/futpro2.0.git

# 7. Subir c√≥digo
git branch -M main
git push -u origin main
